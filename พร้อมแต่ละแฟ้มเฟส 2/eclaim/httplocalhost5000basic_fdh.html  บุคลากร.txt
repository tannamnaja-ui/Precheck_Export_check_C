select position_id,p_name,licenseno,
case when (licenseno_check1 = 'Y'  or licenseno like '-99999%') and licenseno_check2='Y' then 'Y' else 'N' end as licenseno_check,
case when licenseno_check2 = 'Y' or licenseno like '-99999%' then 'Y' else 'N' end as licenseno_check2,
code,name as d_name,cid,
cid_check,pname,case when (pname='' or pname is null or pname not in(select name from pname)) then 'N' else 'Y' end as pname_check,
fname,lname
from(
select position_id,licenseno,p.name as p_name,
case when position_id in(select position_id from position_check ) 
and (licenseno like 'พ%' OR licenseno like 'ว%' OR licenseno like 'ภ%' OR licenseno like 'ท%' OR licenseno like '-%') then 'Y'
else 'N' 
end as licenseno_check1,
case when position_id in(select position_id from position_check ) 
and (licenseno like 'พ.%' OR licenseno like 'ว.%' OR licenseno like 'ภ.%' OR licenseno like 'ท.%' OR licenseno like 'ก%' OR licenseno like 'พป%' OR licenseno like 'พทน%' ) then 'N'
else 'Y' 
end as licenseno_check2,
code,d.name,cid,
case when (cid = '' or cid is null or length(cid)<>13)  then 'N' else 'Y' end as cid_check,pname,fname,lname 
from doctor d
left outer join doctor_position p on p.id=d.position_id 
where d.active='Y'
and d.name not like '%BMS%' and d.name not like '%bms%'
order by position_id,d.name
)tt
