http://localhost:5000/basic_fdh.html  เฉพาะเมนู สัญชาติ nationality ให้ใช้คิรี่นี้ในการแสดงข้อมูล

select n.nationality,n.name,
 n.nhso_code,
 (case when n.nhso_code=p.code and n.name=p.name then 'Y' else 'N' end)as nationality_check 
 from nationality n 
 left outer join provis_nation p on p.code=n.nhso_code
 order by n.nhso_code;

ถ้า ✓ เป็น N ให้คอลั่มข้างหน้า เป็นสีแดงสด
