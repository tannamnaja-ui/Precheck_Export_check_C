เมื่อกดเมนู LABFU
ให้ใช้คำสั่งนี้ในการคิวรี่ข้อมูล

SELECT 
  (SELECT hospitalcode FROM opdconfig LIMIT 1) AS HCODE,
  p.hn AS HN,p.cid,
  CASE WHEN (p.cid IS NULL OR p.cid = '') THEN 'N' ELSE 'Y' END AS PERSON_ID,
  ovst.vstdate,
  CASE WHEN (ovst.vstdate IS NULL) THEN 'N' ELSE 'Y' END AS DATEOPD,
  ovst.vn,
  CASE WHEN (ovst.vn IS NULL) THEN 'N' ELSE 'Y' END AS SEQ,
  sll.sys_lab_code_id,
  CASE WHEN (sll.sys_lab_code_id IS NULL) THEN 'N' ELSE 'Y' END AS LABTEST,
  lo.lab_order_result,
  CASE WHEN (lo.lab_order_result IS NULL) THEN 'N' ELSE 'Y' END AS LABRESULT

FROM
  patient p 
  LEFT JOIN ovst ON ovst.hn = p.hn
  LEFT JOIN lab_head l ON l.vn = ovst.vn
  LEFT JOIN lab_order lo ON lo.lab_order_number = l.lab_order_number
  LEFT JOIN lab_items li ON li.lab_items_code = lo.lab_items_code
  LEFT JOIN sys_lab_link sll ON sll.lab_items_code = li.lab_items_code

WHERE
 ovst.vstdate BETWEEN '2026-03-14' AND '2026-03-30' AND ovst.pttype IN (SELECT pttype.pttype FROM pttype
  WHERE pttype.isuse = 'Y' AND pttype.hipdata_code IN ('UCS', 'WEL'))


'2026-03-14'  and '2026-03-30' วันที่ คือวันที่ ที่เลือกที่หน้าจอ
and pttype in(select pttype from pttype where isuse='Y' and hipdata_code IN ('UCS', 'WEL')) คือสิทธิที่เลือกจากตัวเลือกเท่านั้น
ถ้าไม่มีการเลือกสิทธิ คือเลือกทุกสิทธิ

ถ้า ✓ เป็น N ให้คอลั่มข้างหน้า เป็นสีแดงสด
