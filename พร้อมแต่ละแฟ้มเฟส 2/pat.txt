เมื่อกดเมนู pat 
ให้ใช้คำสั่งนี้ในการคิวรี่ข้อมูล

SELECT 
  (select hospitalcode from opdconfig LIMIT 1) AS "HCODE",
  p.hn AS "HN",p.chwpart,
  CASE WHEN (p.chwpart IS NULL OR p.chwpart = '') THEN 'N' ELSE 'Y' END AS "CHANGWAT" ,
	p.amppart,
  CASE WHEN (p.amppart IS NULL OR p.amppart = '') THEN 'N' ELSE 'Y' END AS "AMPHUR",
	p.birthday,
  CASE WHEN (p.birthday IS NULL) THEN 'N' ELSE 'Y' END AS "BOD",
	p.sex,
  CASE WHEN (p.sex IS NULL OR p.sex = '') THEN 'N' ELSE 'Y' END AS "SEX",
	p.marrystatus,
  CASE WHEN (p.marrystatus IS NULL OR p.marrystatus = '') THEN 'N' ELSE 'Y' END AS "MARRIAGE",
	p.occupation,
  CASE WHEN (p.occupation IS NULL OR p.occupation = '') THEN 'N' ELSE 'Y' END AS "OCCUPA",
	p.nationality,
  CASE WHEN (p.nationality IS NULL OR p.nationality = '') THEN 'N' ELSE 'Y' END AS "NATION",
	p.cid,
  CASE WHEN (p.cid IS NULL OR p.cid = '') THEN 'N' ELSE 'Y' END AS "PERSON_ID",
	Concat(p.fname,' ',p.lname) as "ptname",
  CASE WHEN (Concat(p.fname,' ',p.lname) IS NULL OR Concat(p.fname,' ',p.lname) = ' ') THEN 'N' ELSE 'Y' END AS "NAMEPAT",
	p.pname,
  CASE WHEN (p.pname IS NULL OR p.pname = '') THEN 'N' ELSE 'Y' END AS "TITLE",
	p.fname,
  CASE WHEN (p.fname IS NULL OR p.fname = '') THEN 'N' ELSE 'Y' END AS "FNAME",
	p.lname,
  CASE WHEN (p.lname IS NULL OR p.lname = '') THEN 'N' ELSE 'Y' END AS "LNAME",
  1 AS "IDTYPE"

FROM
  patient P 
	left outer join ovst o on o.hn=p.hn
	where 
o.vstdate between '2025-12-01'  and '2025-12-10'
and o.pttype in(select pttype from pttype where isuse='Y' and hipdata_code in('UCS','WEL'))

 '2025-12-01'  and '2025-12-10' วันที่ คือวันที่ ที่เลือกที่หน้าจอ
and pttype in(select pttype from pttype where isuse='Y' and hipdata_code='OFC') คือสิทธิที่เลือกจากตัวเลือกเท่านั้น
ถ้าไม่มีการเลือกสิทธิ คือเลือกทุกสิทธิ

ถ้า ✓ เป็น N ให้คอลั่มข้างหน้า เป็นสีแดงสด
