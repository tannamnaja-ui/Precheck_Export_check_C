เมื่อกดเมนู IRF
ให้ใช้คำสั่งนี้ในการคิวรี่ข้อมูล

SELECT ip.an,
  r.refer_hospcode AS refer,
  (CASE WHEN r.refer_hospcode <> ''THEN 'Y'ELSE 'N'END) AS check_refer,
  rn.vn AS referin,
  r.vn AS referout,
  (CASE WHEN  (rn.vn IS NOT NULL OR  r.vn IS NOT NULL) THEN 'Y' ELSE 'N'END) AS check_refertype

FROM ipt ip
  LEFT OUTER JOIN referout r ON r.vn = ip.an
  LEFT OUTER JOIN referin rn ON rn.vn = ip.an
 WHERE ip.dchdate between '2026-03-14'  and '2026-03-30'
and ip.pttype in(select pttype from pttype where isuse='Y' and hipdata_code in('UCS','WEL'))
AND Length(r.vn) > 0

'2026-03-14'  and '2026-03-30' วันที่ คือวันที่ ที่เลือกที่หน้าจอ
and pttype in(select pttype from pttype where isuse='Y' and hipdata_code='OFC') คือสิทธิที่เลือกจากตัวเลือกเท่านั้น
ถ้าไม่มีการเลือกสิทธิ คือเลือกทุกสิทธิ

ถ้า ✓ เป็น N ให้คอลั่มข้างหน้า เป็นสีแดงสด
