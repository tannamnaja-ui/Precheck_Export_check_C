SELECT (CASE
    WHEN i.ipt_admit_type_id = '5'
    THEN Concat('ODS', op.an)
    ELSE op.an
  END) AS an,
  ot.name as oper_type,
  op.icd9 AS oper,
  (CASE
    WHEN op.icd9 <> ''
    THEN 'Y'
    ELSE 'N'
  END) AS check_icd9,
  op.opdate AS datein,
  (CASE
    WHEN op.opdate IS NOT NULL
    THEN 'Y'
    ELSE 'N'
  END) AS check_datein,
  op.optime AS timein,
  (CASE
    WHEN op.optime IS NOT NULL AND op.optime <> '00:00:00'
    THEN 'Y'
    ELSE 'N'
  END) AS check_timein,
  op.enddate AS dateout,
  (CASE
    WHEN op.enddate IS NOT NULL
    THEN 'Y'
    ELSE 'N'
  END) AS check_dateout,
  op.endtime AS timeout,
  (CASE
    WHEN op.endtime IS NOT NULL AND op.endtime <> '00:00:00'
    THEN 'Y'
    ELSE 'N'
  END) AS check_timeout,
  d.licenseno AS licenseno,
  (CASE
    WHEN d.licenseno <> ''
    THEN 'Y'
    ELSE 'N'
  END) AS check_licenseno
FROM iptoprt op
  LEFT OUTER JOIN doctor d ON d.code = op.doctor
  LEFT OUTER JOIN ipt i ON i.an = op.an
  LEFT OUTER JOIN oper_type ot on ot.oper_type=op.oper_type 
WHERE i.dchdate BETWEEN '2025-12-01' AND '2025-12-10' AND i.pttype IN (SELECT pttype.pttype FROM pttype
  WHERE pttype.isuse = 'Y' AND pttype.hipdata_code IN ('UCS', 'WEL'))
ORDER BY op.an,
  op.priority


 '2025-12-01'  and '2025-12-10' วันที่ คือวันที่ ที่เลือกที่หน้าจอ
and pttype in(select pttype from pttype where isuse='Y' and hipdata_code IN ('UCS', 'WEL')) คือสิทธิที่เลือกจากตัวเลือกเท่านั้น
