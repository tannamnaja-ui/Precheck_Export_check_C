SELECT ov.vn AS svid,
  ov.vstdate,
  Concat(pa.pname, pa.fname, ' ', pa.lname) AS name,
  rc.debt_id AS invno,
  CASE WHEN rc.debt_id IS NOT NULL THEN 'Y' ELSE '' END AS check_debt_id,
  CASE
    WHEN n.income = '11' THEN 'OP'
    WHEN n.income = '12' THEN 'EC'
    WHEN n.income = '07' THEN 'LB'
    WHEN n.income = '08' THEN 'XR'
    WHEN n.income = '09' THEN 'IV'
    WHEN n.income = '06' THEN 'ZZ'
    ELSE ''
  END AS class,
  CASE WHEN n.income IS NOT NULL THEN 'Y' ELSE '' END AS check_income,
  (SELECT opdconfig.hospitalcode FROM opdconfig LIMIT 1) AS hcode,
  pa.hn,
  pa.cid AS pid,
  ot.ovstist AS typein,
  CASE WHEN ot.ovstist IS NOT NULL THEN 'Y' ELSE 'N' END AS check_typein,
  st.ovstost AS typeout,
  CASE WHEN st.ovstost IS NOT NULL THEN 'Y' ELSE 'N' END AS check_typeout,
  oa.nextdate,
  CASE WHEN oa.nextdate IS NOT NULL THEN 'Y' ELSE '' END AS check_nextdate,
  ov.spclty AS clinic,
  CASE WHEN ov.spclty IS NOT NULL THEN 'Y' ELSE '' END AS check_clinic,
  d.licenseno AS svpid,
  CASE WHEN d.licenseno IS NOT NULL THEN 'Y' ELSE '' END AS check_svpid,
  ov.vsttime AS begdt,
  CASE WHEN ov.vsttime IS NOT NULL THEN 'Y' ELSE '' END AS check_begdt,
  Max(ost.update_datetime) AS enddt,
  CASE WHEN Max(ost.update_datetime) IS NOT NULL THEN 'Y' ELSE '' END AS check_enddt,
  n.icode AS lccode,
  CASE
    WHEN dp.icd9 IS NOT NULL AND dp.icd9 <> '' THEN 'IN'
    WHEN li.tmlt_code IS NOT NULL AND li.tmlt_code <> '' THEN 'LC'
    WHEN di.icd10 IS NOT NULL AND di.icd10 <> '' THEN 'TT'
    ELSE ''
  END AS codeset, 
  CASE
    WHEN dp.icd9 IS NOT NULL AND dp.icd9 <> '' THEN 'Y'
    WHEN li.tmlt_code IS NOT NULL AND li.tmlt_code <> '' THEN 'Y'
    WHEN di.icd10 IS NOT NULL AND di.icd10 <> '' THEN 'Y'
  ELSE ''
  END AS check_codeset,
  n.billcode AS stdcode,
  CASE WHEN n.billcode IS NOT NULL THEN 'Y' ELSE '' END AS check_stdcode ,
  'CareAccount' AS careaccount,
  'TypeServ' AS typeserv,
  'SvCharge' AS svcharge,
  'Completion' AS completion,
  'SvTxCode' AS svtxcode,
  'ClaimCat' AS claimcat,
  CONCAT(ov.vn,':',rc.debt_id) AS vn
FROM ovst ov
  LEFT JOIN patient pa ON pa.hn = ov.hn
  LEFT JOIN rcpt_debt rc ON rc.vn = ov.vn
  LEFT JOIN ovstist ot ON ot.ovstist = ov.ovstist
  LEFT JOIN oapp oa ON oa.vn = ov.vn
  LEFT JOIN (SELECT DISTINCT opitemrece.vn,opitemrece.icode FROM opitemrece) oi ON oi.vn = ov.vn
  LEFT JOIN ovstost st ON st.ovstost = ov.ovstost
  LEFT JOIN ovst_service_time ost ON ost.vn = ov.vn
  LEFT JOIN nondrugitems n ON n.icode = oi.icode
  LEFT JOIN (SELECT ovstdiag.vn,ovstdiag.icd10,ovstdiag.doctor,ovstdiag.ovst_diag_id,
    Row_Number() OVER (PARTITION BY ovstdiag.vn ORDER BY ovstdiag.ovst_diag_id) AS rn
  FROM ovstdiag) di ON di.vn = ov.vn AND di.rn = 1
  LEFT JOIN (SELECT doctor_operation.ovst_diag_id,doctor_operation.icd9,
    Row_Number() OVER (PARTITION BY doctor_operation.ovst_diag_id ORDER BY doctor_operation.ovst_diag_id) AS rn
  FROM doctor_operation) dp ON dp.ovst_diag_id = di.ovst_diag_id AND dp.rn = 1
  LEFT JOIN doctor d ON d.code = di.doctor
  LEFT JOIN lab_items li ON li.icode = n.icode
  LEFT JOIN drugitems dt ON dt.icode = oi.icode
WHERE ov.vn IN ('680721000009')
GROUP BY ov.vn,
  ov.vstdate,
  Concat(pa.pname, pa.fname, ' ', pa.lname),
  rc.debt_id,
  pa.hn,
  pa.cid,
  ot.ovstist,
  st.ovstost,
  oa.nextdate,
  ov.spclty,
  d.licenseno,
  ov.vsttime,
  n.icode,
  n.billcode,
  li.lab_items_code,
  dt.icode,
  n.name,
  dt.name,
  dp.icd9,
  li.tmlt_code,
  di.icd10
